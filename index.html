Leave Management Schema Setup
Run this SQL once in Supabase to enable the comprehensive leave management module (accruals, balances, audit log):
CREATE TABLE IF NOT EXISTS leave_balances (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_name text NOT NULL UNIQUE,
  department text,
  employment_start_date date,
  accrued_days numeric DEFAULT 0,
  manual_adjustment numeric DEFAULT 0,
  used_annual numeric DEFAULT 0,
  used_family numeric DEFAULT 0,
  used_sick numeric DEFAULT 0,
  used_unpaid numeric DEFAULT 0,
  last_accrual_month text,
  updated_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS leave_accrual_log (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_name text NOT NULL,
  accrual_month text NOT NULL,
  days_added numeric NOT NULL DEFAULT 1.5,
  balance_before numeric,
  balance_after numeric,
  note text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE leave_records ADD COLUMN IF NOT EXISTS days_requested numeric;
ALTER TABLE leave_records ADD COLUMN IF NOT EXISTS reason text;
ALTER TABLE leave_records ADD COLUMN IF NOT EXISTS manager_notes text;
ALTER TABLE leave_records ADD COLUMN IF NOT EXISTS approved_by text;
ALTER TABLE leave_records ADD COLUMN IF NOT EXISTS approved_at timestamptz;
ALTER TABLE leave_balances ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "allow_all" ON leave_balances FOR ALL USING (true) WITH CHECK (true);
ALTER TABLE leave_accrual_log ENABLE ROW LEVEL SECURITY;
CREATE POLICY IF NOT EXISTS "allow_all" ON leave_accrual_log FOR ALL USING (true) WITH CHECK (true);
Attendance Log Table Setup
Run this SQL once in Supabase to enable QR code clock-in/out:
CREATE TABLE IF NOT EXISTS attendance_log (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  staff_name text NOT NULL,
  token text,
  type text NOT NULL CHECK (type IN ('check-in','check-out')),
  date date NOT NULL,
  time_str text,
  logged_at timestamptz DEFAULT now()
);
ALTER TABLE attendance_log ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON attendance_log FOR ALL USING (true) WITH CHECK (true);
CREATE INDEX IF NOT EXISTS att_date_idx ON attendance_log(date);
CREATE INDEX IF NOT EXISTS att_staff_idx ON attendance_log(staff_name);
Micros Imports Table Setup
Run this SQL once in your Supabase SQL editor to enable Micros report uploads:
CREATE TABLE IF NOT EXISTS micros_imports (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  date date NOT NULL,
  shift text NOT NULL DEFAULT 'Full Day',
  items jsonb,
  total_sales numeric DEFAULT 0,
  total_qty int DEFAULT 0,
  total_voids int DEFAULT 0,
  uploaded_by text,
  uploaded_at timestamptz DEFAULT now(),
  UNIQUE(date, shift)
);
ALTER TABLE micros_imports ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON micros_imports FOR ALL USING (true) WITH CHECK (true);
Spot Check Table Setup
Run this SQL in your Supabase SQL editor to enable the Daily Spot Check feature:
CREATE TABLE IF NOT EXISTS spot_checks (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  date date NOT NULL UNIQUE,
  submitted_by text,
  submitted_at timestamptz DEFAULT now(),
  items jsonb,
  flagged_count int DEFAULT 0,
  completed_count int DEFAULT 0,
  total_items int DEFAULT 15
);
ALTER TABLE spot_checks ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON spot_checks FOR ALL USING (true) WITH CHECK (true);

ALTER TABLE payroll ADD COLUMN IF NOT EXISTS breakage numeric DEFAULT 0;

ALTER TABLE payroll DROP CONSTRAINT IF EXISTS payroll_staff_name_month_key;
